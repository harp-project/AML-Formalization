Low-level proof, using only the proof system
Lemma ex1_low : ∀ {Σ : Signature} (Γ : Theory) (A B : Pattern),
  well_formed A = true →
  well_formed B = true →
  Γ ⊢i (A ---> B ---> (A and B))
  using BasicReasoning.

Proof using proof mode
Lemma ex1_pm : ∀ {Σ : Signature} (Γ : Theory) (A B : Pattern),
  well_formed A = true →
  well_formed B = true →
  Γ ⊢i (A ---> B ---> (A and B))
  using BasicReasoning.

Lemma ex1_coq (A B : Prop) :
  A → B → A ∧ B.

Lemma ex1_pm2 : ∀ {Σ : Signature} (Γ : Theory) (A B : Pattern),
  well_formed A = true →
  well_formed B = true →
  Γ ⊢i (A ---> B ---> (A and B))
  using BasicReasoning.

    |}.
